import java.util.Arrays;
import java.util.Scanner;

public class Main {
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int N = sc.nextInt();
		int M = sc.nextInt();
		int[] arr = new int[N];
		for (int i = 0; i < N; i++) {
			arr[i] = sc.nextInt();
		}
		Arrays.sort(arr);
		int start = 1;
		int end = N - 1;
		int answer = 0;

		while (start <= end) {
			int mid = (start + end) / 2;
			if (arr[mid] == M) {
				answer = mid;
				break;
			} else if (arr[mid] > M) {
				end = mid - 1;
			} else {
				start = mid + 1;
			}
		}
		System.out.println(answer + 1);
	}
}

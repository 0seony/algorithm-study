import java.util.Scanner;

public class Main {
	static String answer = "NO";
	static int[] number;
	static boolean flag;
	static int N, sum = 0;

	static void DFS(int L, int cnt) {
		if(flag) return;
		if(cnt>sum/2) return;
		if(L==N) {
			if((sum-cnt)==cnt) {
				answer = "YES";
				flag = true;
			}

		} else {
			DFS(L+1, cnt+number[L]);
			DFS(L+1, cnt);
		}
	}

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		N = sc.nextInt();
		number = new int[N];
		for (int i = 0; i < N; i++) {
			number[i] = sc.nextInt();
			sum += number[i];
		}
		DFS(0, 0);
		System.out.println(answer);
	}
}

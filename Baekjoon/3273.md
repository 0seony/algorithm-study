import java.util.ArrayList;
import java.util.Arrays;
import java.util.Scanner;

public class Main {
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int n = sc.nextInt();
		int[] arr = new int[n];
		for (int i = 0; i < n; i++) {
			arr[i] = sc.nextInt();
		}
		int x = sc.nextInt();
		Arrays.sort(arr);
		int lt = 0;
		int rt = n - 1;
		int answer = 0;
		while (lt < rt) {
			int sum = arr[lt] + arr[rt];
			if (sum > x) {
				rt--;
			} else if (sum < x) {
				lt++;
			} else {
				answer++;
				lt++;
				rt--;
			}
		}
		System.out.println(answer);
	}

}

import java.util.Scanner;

public class Main {
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);

		int[] arr = new int[n];

import java.util.Scanner;

public class Main {
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int N = sc.nextInt();
		int M = sc.nextInt();
		int[] money = new int[N];
		int start = 0;
		int end = 0;
		for (int i = 0; i < N; i++) {
			money[i] = sc.nextInt();
			start = Math.max(start, money[i]);
			end += money[i];
		}
		while (start <= end) {
			int mid = (start + end) / 2;
			int cnt = 1;
			int k = mid;
			for (int i = 0; i < N; i++) {
				if (k < money[i]) {
					cnt++;
					k = mid;
				}
				k-=money[i];
			}
			if (cnt <= M) {
				end = mid - 1;
			} else {
				start = mid + 1;
			}
		}
		System.out.println(start);
	}
}

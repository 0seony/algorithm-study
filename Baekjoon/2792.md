import java.util.Scanner;

public class Main {
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int N = sc.nextInt();
		int M = sc.nextInt();
		int lt = 1; // μ§ν¬μ‹¬ μµμ†
		int rt = 0; // μ§ν¬μ‹¬ μµλ€
		int[] color = new int[M];
		for (int i = 0; i < M; i++) {
			color[i] = sc.nextInt();
			rt = Math.max(rt, color[i]);
		} // μ…λ ¥

		int answer = 0;
		while (lt <= rt) {
			int mid = (rt + lt) / 2;
			int people = 0;
			for (int idx = 0; idx < color.length; idx++) {
				people += Math.ceil((double) color[idx] / mid);
			}
			if (people > N) {
				lt = mid + 1;
			} else if (people <= N) {
				answer = mid;
				rt = mid - 1;
			}
		}
		System.out.println(answer);
	}
}

import java.util.Scanner;

public class Main {
	static int N, M;
	static int[] print;
	static int[] visited;

	static void DFS(int L, int num) {
		if (L == M) {
			for (int i : print) {
				System.out.print(i + " ");
			}
			System.out.println();
		} else {
			for (int i = num; i <= N; i++) {
				if (visited[i] != 1) {
					visited[i] = 1;
					print[L] = i;
					DFS(L + 1, i + 1);
					visited[i] = 0;
				}
			}
		}
	}

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		N = sc.nextInt();
		M = sc.nextInt();
		print = new int[M];
		visited = new int[N + 1];
		DFS(0, 1);
	}
}

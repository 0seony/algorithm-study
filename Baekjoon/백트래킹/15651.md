import java.util.Scanner;

public class Main {
	static int N, M;
	static int[] result;
	static StringBuffer sb;

	public static void DFS(int L) {
		if (L == M) {
			for (int i : result) {
				sb.append(i + " ");
			}
			sb.append("\n");
			return;
		}
		for (int i = 1; i <= N; i++) {
			result[L] = i;
			DFS(L + 1);
		}

	}

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		N = sc.nextInt();
		M = sc.nextInt();
		result = new int[M];
		sb = new StringBuffer();
		DFS(0);
		System.out.println(sb);

	}

}
